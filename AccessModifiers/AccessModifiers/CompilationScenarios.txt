Will the following codes compile? If not, explain the reason.

1. private int _age; 
   public int Age { 
   get { return Age;} 
   set { Age = value;} 
   }
   No this code will not compile because we have returned the property "Age" instead of backing field "_age".

2. public readonly string RegistrationNumber; 
   public void Update () { 
   RegistrationNumber = "PAT1002"; 
   }
   No this code will not compile. A readonly field can only be assigned during declaration or inside the constructor. 
   Assigning it inside the Update() method causes a compile-time error.

3. public class Patient { 
   public const string HospitalCode; 
   public Patient () { 
   HospitalCode = "HMS001"; 
   }
   No this code will not compile. A const variable must be initialized when it is declared. 
   It cannot be assigned inside a constructor because a constant value is fixed at compile time.

4. public class Patient { 
   private int _age; 
   } 
   Patient patient = new Patient (); 
   patient ._age = 25;
   No thsi code will not compile because _age has "private" access modifier. It cannot be called outside the class.

5. public class Person { 
   protected string Mobile = "9876543210"; 
   } 
   Person person = new Person ();
   No this code will not compile because "protected" access modifier does not allow access outside the same class.

6. public class Student
   {
        private int _marks;
    }
    Student s = new Student();
    Console.WriteLine(s._marks);
    No thsi code will not compile because _marks is private and cannot be accessed outside the same class.

7. public class Employee
    {
        public readonly int Id = 101;
    }
    Employee e = new Employee();
    e.Id = 102;
    No this code will not compile because a readonly field cannot be modified after object creation.

8. public class Company
    {
        public const string Name = "Microsoft";
    }
    Company.Name = "Google";
    No this code will not compile because a const value cannot be changed.

9. public class Person
    {
        protected string Name = "John";
    }
    Person p = new Person();
    Console.WriteLine(p.Name);
    No this code will not compile because "protected" access modifier does not allow access outside the same class.

10. public class Student
    {
        public int Age
        {
            get;
            private set;
        }
    }
    Student s = new Student();
    s.Age = 20;
    No this code will not compile because setter is private and cannot be accessed outside the class.

11. public class Person
    {
        public virtual void Show()
        {
    }
    }
    public class Student : Person
    {
        public override void Show()
        {
    }
    }
    Yes this code will compile because a virtual method can be overridden in the derived class.

12. sealed class Vehicle
    {
    }
    class Car : Vehicle
    {
    }
    No this code will not compile because a sealed class cannot be inherited.

13. public class Employee
    {
        public static int Count = 0;
    }
    Console.WriteLine(Employee.Count);
    Yes this code will compile.

14. public class Test
    {
        public int Age;
    }
    Console.WriteLine(Test.Age);
    No this code will not compile because Age is an instance field, so an object is required.

15. public class Person
    {
        public string Name { get; set; }
    }
    Person p = new Person();
    p.Name = "Sam";
    Yes, this code will compile.
